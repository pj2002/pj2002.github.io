第6关：三角形的面积和周长

import math

a = float(input("请输入三角形的边a:"))

b = float(input("请输入三角形的边b:"))

c = float(input("请输入三角形的边c:"))

p=(a+b+c)/2

if a+b<=c or a+c<=b or b+c<=a:

print("无法构成三角形！")

else:

print("三角形的周长=%.1f，面积=%.1f"%(p*2,math.sqrt(p*(p-a)*(p-b)*(p-c))))
